```python
# классом пользуется 100 модулей по всему проекту, видоизменять функциональность модуля тяжело,
# по сути считаем его закрытым.
class FileReader:
    """Обеспечивает удобный интерфейс для чтения файлов"""


# Появилась потребность в чтении файлов, содержащих json, с преобразованием в python/pydantic объект
# Расширим функциональность FileReader. Класс JsonFileReader будет считаться открытым, так как находится
# в разработке - можно расширять сколь душе угодно.
class JsonFileReader(FileReader):
    """Обеспечивает интерфейс для чтения json-файлов с преобразованием содержимого в python объект"""


# Аналогично при потребности в чтении yaml файлов создаем открытый к расширению модуль.
class YamlFileReader(FileReader):
    """Обеспечивает интерфейс для чтения yaml-файлов с преобразованием содержимого в python объект"""
```
